            const s = document.getElementById('select-bf-item');

        async function salvarProduto() {
            const d = { 
                name: document.getElementById('p-nome').value, 
                price: document.getElementById('p-preco').value, 
                url: document.getElementById('p-link').value, 
                category: document.getElementById('p-categoria-select').value,
                img: document.getElementById('p-imgs').value, 
                description: document.getElementById('p-desc').value,
                personaliza: document.getElementById('p-personaliza').value,
                url_extra2: document.getElementById('p-off').value, // USANDO ESTA COLUNA PARA O % OFF
                blackfriday: "false" 
            };
            await _supabase.from('produtos').insert([d]);
            location.reload();
        }
        function renderGrid(l) {
            document.getElementById('grid').innerHTML = l.map(p => `
                <div class="card" onclick="verDetalhes('${p.id}')">
                    ${p.blackfriday == "true" ? '<span class="badge-bf">OFERTA</span>' : ''}
                    ${p.url_extra2 ? `<div class="ribbon-off">${p.url_extra2}% OFF</div>` : ''}
                    <div class="img-box"><img src="${p.img.split(',')[0]}"></div>
                    <h3 style="margin-top:10px; font-size:14px;">${p.name}</h3>
                    <p style="color:var(--cyan); font-weight:bold;">R$ ${p.price}</p>
                </div>
            `).join('');
        }
        function verDetalhes(id) {
            const p = produtosFull.find(x => x.id == id);
            document.getElementById('m-nome').innerText = p.name;
            document.getElementById('m-preco').innerText = `R$ ${p.price}`;
            document.getElementById('m-desc').innerText = p.description;
            document.getElementById('m-img-main').src = p.img.split(',')[0];

            document.getElementById('modal-produto').style.display = 'flex';
        }
        // FUNÇÕES DE SISTEMA
        async function salvarDestaqueBF() {
            const id = document.getElementById('select-bf-item').value;
            const dataFim = document.getElementById('data-bf-fim').value;
            await _supabase.from('produtos').update({ blackfriday: "false" }).neq('id', '0');
            if(id) await _supabase.from('produtos').update({ blackfriday: "true", url_extra: dataFim }).eq('id', id);
            location.reload();
        }
        function checkBF() {
            const p = produtosFull.find(x => x.blackfriday == "true");
            if(p && p.url_extra && new Date(p.url_extra).getTime() > new Date().getTime()) {
                document.getElementById('bf-highlight').style.display = 'flex';
                document.getElementById('bf-title').innerText = p.name;
                document.getElementById('bf-img').src = p.img.split(',')[0].trim();
                document.getElementById('bf-link').onclick = () => window.open(p.url, '_blank');
                setInterval(() => {
                    const dist = new Date(p.url_extra).getTime() - new Date().getTime();
                    if(dist > 0) {
                        const h = Math.floor(dist/3600000); const m = Math.floor((dist%3600000)/60000); const s = Math.floor((dist%60000)/1000);
                        document.getElementById('countdown').innerText = `${h}h ${m}m ${s}s`;
                    }
                }, 1000);
            }
        }

            renderGrid(produtosFull.filter(p => p.name.toLowerCase().includes(t)));
        }
